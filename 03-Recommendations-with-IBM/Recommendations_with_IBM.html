def calculate_stats(data):
    ''' 
    Plots a graph of a distribution with and without outliers
     Args:
       interactions: pandas series or array to calculate statistics
     Returns:
     mean float: mean of the interactions array  
     median float: mean of the interactions array 
     non_outliers array: array of the non outliers in the input array 
    
    '''
    percentile_25 = np.percentile(data, 25)
    percentile_75 = np.percentile(data, 75)
    median = np.percentile(data, 50)
    mean = np.mean(data)
    
    IQR = percentile_75 - percentile_25
    lower_cutoff = percentile_25 - 1.5*IQR
    upper_cutoff = percentile_75 + 1.5*IQR
    
    non_outliers = data[(data > lower_cutoff) & (data < upper_cutoff)]
    
    return  mean, median, non_outliers


def plot_distribution(data, title, xlabel, ylabel):
    ''' 
    Plots a graph of a distribution with and without outliers
     Args:
      data: pandas series or array to be plotted
      title str: Title of the plot to be made
      xlabel str: X label of the plot to be made
      ylabel str: Y labelof the plot to be made
     Returns: None
    '''
    fontdict = {'fontsize': 10}   
    
    mean, median, non_outliers = calculate_stats(data)
    
    fig, ax = plt.subplots(1,2, figsize=(12,6))
    sns.distplot(data, ax=ax[0], kde=False)
    sns.distplot(non_outliers, ax=ax[1], bins=range(0,non_outliers.max(),1), kde=False, rug=True);

    ax[1].axvline(median, ymax=0.95, color='forestgreen')
    ax[1].axvline(mean, ymax=0.95, color='red')
    ax[1].text(median+0.2, 1000,f'<-median: \n {median:.2f}')
    ax[1].text(mean+0.2, 1000 ,f'<-mean: \n {mean:.2f}')
    
    ax[0].set_title(title + ' with Outliers', fontdict=fontdict)
    ax[0].set_xlabel(xlabel)
    ax[0].set_ylabel(ylabel)
    
    ax[1].set_title(title+' without Outliers', fontdict=fontdict)
    ax[1].set_xlabel(xlabel)
    ax[1].set_ylabel(ylabel)

def find_similar_users(user_id, user_item=user_item):
    '''
    INPUT:
    user_id - (int) a user_id
    user_item - (pandas dataframe) matrix of users by articles: 
                1's when a user has interacted with an article, 0 otherwise
    
    OUTPUT:
    similar_users - (list) an ordered list where the closest users (largest dot product users)
                    are listed first
    
    Description:
    Computes the similarity of every pair of users based on the dot product
    Returns an ordered list
    
    '''
    # compute similarity of each user to the provided user
    dot_product_matrix = user_item.loc[user_id].dot(user_item.T)

    # sort by similarity
    sorted_matrix = dot_product_matrix.sort_values(ascending=False)
    
    # create list of just the ids
    most_similar_users = list(sorted_matrix.index.values)
   
    # remove the own user's id
    most_similar_users.remove(user_id)
        
    return most_similar_users # return a list of the users in order from most to least similar

def user_user_recs(user_id, m=10):
    '''
    INPUT:
    user_id - (int) a user id
    m - (int) the number of recommendations you want for the user
    
    OUTPUT:
    recs - (list) a list of recommendations for the user
    
    Description:
    Loops through the users based on closeness to the input user_id
    For each user - finds articles the user hasn't seen before and provides them as recs
    Does this until m recommendations are found
    
    Notes:
    Users who are the same closeness are chosen arbitrarily as the 'next' user
    
    For the user where the number of recommended articles starts below m 
    and ends exceeding m, the last items are chosen arbitrarily
    
    '''
    # Your code here
    recs = []
    seen_article_ids = get_user_articles(user_id)[0]
    similar_user_ids = find_similar_users(user_id)
    
    for similar_user_id in similar_user_ids:
        user_recs = get_user_articles(similar_user_id)
        unseen_article_ids = np.setdiff1d(user_recs[0], seen_article_ids)
        new_recs_ids = list(np.setdiff1d(unseen_article_ids, recs))
        recs.extend(new_recs_ids)
        if len(recs) >= m:
            break
    recs = recs[:m]
    return recs # return your recommendations for this user_id

def get_top_sorted_users(user_id, df=df, user_item=user_item):
    '''
    INPUT:
    user_id - (int)
    df - (pandas dataframe) df as defined at the top of the notebook 
    user_item - (pandas dataframe) matrix of users by articles: 
            1's when a user has interacted with an article, 0 otherwise
    
            
    OUTPUT:
    neighbors_df - (pandas dataframe) a dataframe with:
                    neighbor_id - is a neighbor user_id
                    similarity - measure of the similarity of each user to the provided user_id
                    num_interactions - the number of articles viewed by the user
                    
    Other Details - sort the neighbors_df by the similarity and then by number of interactions where 
                    highest of each is higher in the dataframe
     
    '''
    # Your code here
    dot_product_matrix = (user_item.loc[user_id]).dot(user_item.T).to_frame()
    
    #Views counts on article by each user
    article_views_df = df.groupby('user_id')['article_id'].count().to_frame()
    
    similarity_df = pd.merge(dot_product_matrix, article_views_df, left_index=True, right_index=True).reset_index()
    
    similarity_df = similarity_df.rename(columns={user_id:'similarity',
                                                'article_id':'num_interactions', 
                                                'user_id':'neighbor_id'})
    
    similarity_df = similarity_df[~similarity_df['neighbor_id'].eq(user_id)]
    
    neighbors_df = similarity_df.sort_values(['similarity','num_interactions'],ascending=False).reset_index(drop=True)
    
    return neighbors_df # Return the dataframe specified in the doc_string

def tokenize(text):
    '''
    INPUT:
    text - (str) a text string to tokenize
    
    OUTPUT:
    tokens - (list) a of tokenized words
    
    Description:
    Strips punctation, stopwords, normalizes a text string and breaks it up into a list of Lemmatized word tokens.
    '''
    text = re.sub(r'[^a-zA-Z0-9]',' ',text.lower())
    text = re.sub(r' +[0-9]+',' ',text)
    tokens = word_tokenize(text)
    
    stop_words = stopwords.words('english')
    lemmatizer = WordNetLemmatizer()
    
    tokens = [lemmatizer.lemmatize(word) for word in tokens if word not in stop_words]
    tokens = [lemmatizer.lemmatize(word, pos='v') for word in tokens]
    
    return tokens


def get_similar_articles(article_id, df_content=df_content, column_name='doc_full_name', return_matrix=False):
    '''
    INPUT:
    article_id - (str): Expects an article_id like '1.0' or '100.0' as a string.
    df_content - (pandas dataframe) df_content as defined at the top of the notebook
    column_name - (string) he column name from the df_content dataframe to use in calculating similarity
    
    return_matrix - (bool) If true, returns the similarity_mat of article-article similarity as well as the similar articles matrix.
                    
        
    OUTPUT:
    article_similarity - (pandas dataframe) a row containing article similarity ratings with article ids as column names
    similarity_mat - (pandas dataframe) (optional if return_matrix=True):
                   A matrix of article-article similarities. 
                   Every article has the most similar article as itself across the diagonal.
                   The index and columns defined as the respective article id's. 

    Description:
    Takes an article id and a dataframe of content.
    Cleans and returns similar articles  to the article_id calculated with cosine distance on TF-IDF.
    '''

    vect = TfidfVectorizer(tokenizer=tokenize, ngram_range=(1,2))
    X_transf = vect.fit_transform(df_content[column_name])
    
    article_ids = df_content['article_id'].astype(float).astype(str).values
    
    similarities = cosine_similarity(X_transf)
    
    similarity_mat = pd.DataFrame(similarities,
                                  columns=article_ids,
                                  index=article_ids)

    article_similarity = similarity_mat[similarity_mat.index == article_id].drop(columns=article_id).T
    
    article_similarity = article_similarity.rename(columns={article_id:'similarity'})
    
    if return_matrix:
        return article_similarity, similarity_mat
    
    return article_similarity


def get_similar_to_new_article(article_content, df_content=df_content, column_name='doc_full_name'):
    '''
    INPUT:

    df_content - (pandas dataframe) df_content as defined at the top of the notebook

    OUTPUT:

    The index and columns defined as the respective article id's.

def sort_article_similarities(article_similarity, df=df, sortby=['similarity','num_interactions']):
    '''
    INPUT:
    article_similarity - (pandas dataframe) df articles similar to article_id by some numerical similarity
    df - (pandas dataframe) with article_id, title, user_id columns
    sortby  - (list) a list detailing how to sort article_similarity by interactions, similarity, or both.
            Accepts: ['num_interactions','similarity'] will sort by interactions, then similarity
                     ['similarity','num_interactions'] would sort by similarity, then interactions
                     ['num_interactions'] would sort by interactions only
                     ['similarity'] would sort by similarity only
    
    OUTPUT:
    sorted_df - (pandas dataframe) a dataframe with:
                    article_id - is a neighboring article id
                    similarity - measure of the similarity of each article to the provided article_id
                    num_interactions - the number of times an articles  was viewed by a user
                    
    Other Details - sorts the input df by the similarity and then by number of interactions where the 
                    highest of each is higher in the dataframe. Where the interaction data is missing,
                    it sets interaction to zero for athat particular article.
    '''
    
    #get ids in the index and add article id if exists
    article_idxs = article_similarity.index.unique() 

    #check if ids in the index are present in the content df
    intersect_ids = np.intersect1d(df['article_id'].unique(), article_idxs)

    #only some articles are in the interaction data for these we append interaction data
    article_interactns = df.groupby('article_id')['article_id'].count()
    article_interactns = article_interactns[article_interactns.index.isin(intersect_ids)] #.to_frame()

    #set articles with missing interaction data to zero
    missing = np.setdiff1d(article_similarity.index.unique(), article_interactns.index.unique())
    article_interactns = article_interactns.append(pd.Series(np.zeros(len(missing)),index=missing)).to_frame()

    similarity_df = pd.merge(article_similarity, article_interactns,
                             left_index=True, right_index=True).reset_index()

    similarity_df = similarity_df.rename(columns={'article_id':'num_interactions',
                                                  'index':'article_id',
                                                  0:'num_interactions'})
    
    sorted_df = similarity_df.sort_values(sortby,ascending=False).reset_index(drop=True)       

    return sorted_df

def estimate_recs(user_id, user_item=user_item_train, u=u_train, s=s_train, vt=vt_train, num_feats=None):
    '''
    INPUT:
    user_id - (int) a user id
    user_item_train - a user-item matrix of the training dataframe 
                      (unique users for each row and unique articles for each column)
    u - a user-latent features matrix output from fitting svd on the user_item matrix
    s - a latent_features-latent_features matrix output from fitting svd on the user_item matrix
    vt - a latent_features-articles matrix output from fitting svd on the user_item matrix
    num_feats - (int) the number of latent features to use in making a prediction 
    
    OUTPUT:
    user_recs - (numpy array) a numpy of possible article interactions for the user
    
    Description:
    Makes estimations for possible article interactions by a user given their current interactions 
    '''
    user_idx = np.where(user_item.index == user_id)[0][0]
    if num_feats:
        n_feats = num_feats
    else:
        n_feats = min(u.shape[1],vt.shape[0])
    u_nw, s_nw, vt_nw = u[user_idx,:n_feats], np.diag(s[:n_feats]), vt[:n_feats,:]
    
    user_recs = np.around(np.dot(np.dot(u_nw, s_nw), vt_nw))

    return user_recs

PART A: ON SVD RESULTS.


From the charts above, we can see that:


    

  1. Our training accuracy rises as the number of latent features rises. This is the generally preferred behaviour.
    

    2. 

  2. However, our test accuracy starts at a very high number with even the smallest of latent features. A number of things can explain this:
    

      

    • Small Test Set: We have a small number of estimates we making predictions for in the test set. We are only making recommendations for only 20 users in the test set.
      

      • 

    • Imbalanced Classes: We should note that our accuracy metric is working on a sparse matrix with a large number of zeros and a small number of ones. The is a class imbalance problem and could cause a number of problems. For example, if we have 95 0's and 5 1's in our dataset. If our model predicts 100% zeros, we end up with a 95% accuracy, despite never predicting one class. We could end up with a high bias model that overrecommends the more dominant class and this in leads us to overestimating performance.
      

      • 

    

    3. 



Solutions:
There are a number of ways to solve these issues. We can use Resampling techniques such as 'Oversampling' and Undersampling' to bring more of our underrepresented classes into our training/test data.


We can also change metrics from 'Accuracy' to some other better fitting performance metric like the Precisio, Recall or the F1 Score depending on our goals.


We can also work towards gathering more data and should perhaps running the test again when we have done so.















PART B MAKING RECOMMENDATIONS. 
Given all of the above ways of making recommendations, we still cannot tell if users appreciate or find the articles we recommend useful. Unless a user opts in to interact with a recommendation, then we cannot tell if it is useful.


To determine if users find these useful, there are a number of options to take.


    

  • Online Surveys: 
One way we can get feedback is by directly asking the users opinion if they found the article/recommendation useful. This can be done in the manner of a small onscreen survey with minimal options across different users.
    • 



    

  • User Testing: 
Another way to get feedback on performance is through organized user testing where a representative group of users are brought in to test the system over some time and report their experiences. More detaild feedback can be extracted from the users here. Although, this method is often expensive.
    • 



    

  • Experiments/ A/B Testing:
We can also run an experiments such as an A/B test. We can send different articles, generated by different algorithms (2 or more) to different groups (2 or more) see which one performs better. We can check performance on some metrics like the average Click through Rate to the performance of a baseline model. For a baseline model the most suitable model is perhaps the "most popular articles" or Ranked Recommendation algorithm.
